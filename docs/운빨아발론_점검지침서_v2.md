## 2. 크리티컬 버그 수정

### 🔴 BUG-01: `invincibleTimer` 이중 감소 — 무적 시간이 절반만 작동

**원인**: `Hero.update()`와 `Entity.update()` 두 곳 모두에서 `invincibleTimer`를 감소시킴. Hero 인스턴스는 1프레임에 2번 감소 → 실제 지속시간이 절반.

```javascript
// Hero.update() 안
if(this.invincibleTimer > 0) this.invincibleTimer -= dt;  // 감소 1회
...
super.update(dt);  // Entity.update() 호출 → 감소 2회 (return 전에!)
```

```javascript
// Entity.update() 안
if(this.invincibleTimer>0) {
    this.invincibleTimer-=dt;  // 2번째 감소
    this.vx=0; this.vy=0;
    return;
}
```

**영향**: Archer 블링크 무적 0.3s→0.15s, Zhonya 2.5s→1.25s, 아발론 대검 무적 3s→1.5s

**수정**: `Hero.update()` 내 별도 감소를 **삭제**. Entity.update()에서만 처리.

```javascript
// Hero.update() 내 아래 줄 삭제
// if(this.invincibleTimer > 0) this.invincibleTimer -= dt;  ← 이 줄 제거

// Entity.update() 내 invincible 블록에 stunTimer도 함께 감소 추가
if(this.invincibleTimer>0) {
    this.invincibleTimer -= dt;
    if(this.stunTimer > 0) this.stunTimer -= dt;  // ★ 추가: 스턴도 같이 흘러야 함
    this.vx=0; this.vy=0;
    return;
}
```

### 🔴 BUG-04: `vampireAura` 패시브가 `applyRawDamage` 우회 — 킬 크레딧 없음, 음수 HP 가능

**원인**: `updatePassives()` 내 vampireAura 처리가 `e.hp -= drain`으로 직접 HP를 깎음. 이로 인해:
- isDead 체크 없이 적이 사망할 수 있음 (isDead가 true가 되지 않고 음수 HP 잔류)
- 킬 크레딧이 attacker에게 부여되지 않음
- invincible 체크 없음

```javascript
// 현재 코드 (버그)
let pVamp = this.passiveSkills['vampireAura'] || 0;
if(pVamp > 0) {
    let drain = pVamp * 5 * dt;
    entities.forEach(e => {
        if(e.faction !== this.faction && !e.isDead && dist(this, e) <= 150) {
            e.hp -= drain;  // ← 직접 HP 조작, 사망 체크 없음!
            this.hp = Math.min(this.maxHp, this.hp + drain);
        }
    });
}
```

**수정**:

```javascript
let pVamp = this.passiveSkills['vampireAura'] || 0;
if(pVamp > 0) {
    this._vampireAuraTimer = (this._vampireAuraTimer || 0) + dt;
    if(this._vampireAuraTimer >= 0.3) {  // 0.3초마다 틱
        this._vampireAuraTimer = 0;
        let tickDrain = pVamp * 5 * 0.3;  // 0.3초 분량 피해
        let totalHeal = 0;
        entities.forEach(e => {
            if(e.faction !== this.faction && !e.isDead && dist(this, e) <= 150) {
                let dealt = e.applyRawDamage(tickDrain, this);  // ★ 정상 피해 처리
                totalHeal += dealt;
                spawnParticles(e.x, e.y, '#f43f5e', 2, 40, 0.25);
            }
        });
        if(totalHeal > 0) {
            this.hp = Math.min(this.maxHp, this.hp + totalHeal * 0.5);
        }
    }
}
```

### 🔴 BUG-05: `grrr` 캐릭터 머리/귀/눈이 좌표 0,0(화면 좌상단)에 그려짐

**원인**: `drawBlockyHero()` 내 grrr 블록에서 `drawBody()`는 `x, y` 절대좌표를 사용하는데, 사자 머리/갈기/귀/눈은 좌표 `0, -r*숫자`로 그려짐. ctx는 캐릭터 위치로 translate되지 않은 상태라 전부 화면 원점 근처에 그려짐.

```javascript
// 현재 코드 (버그)
} else if(type === 'grrr') {
    drawBody('#d97706', '#92400e', '#78350f');
    ctx.arc(0, -r*0.6, r*0.9, ...);  // ← 0이 아니라 x를 써야 함!
    ctx.arc(0, -r*0.6, r*0.6, ...);  // ← 마찬가지
    ctx.arc(-r*0.5, -r*1.1, r*0.25, ...);  // ← -r*0.5가 아니라 x-r*0.5
```

**수정**: grrr 블록 전체를 아래로 교체:

```javascript
} else if(type === 'grrr') {
    drawBody('#d97706', '#92400e', '#78350f');
    
    // ★ 모든 좌표를 x, y 기준으로 수정
    // 사자 갈기 (풍성한 외곽)
    ctx.fillStyle = '#b45309';
    ctx.beginPath();
    ctx.arc(x, y - r*0.6, r*0.9, 0, Math.PI*2);
    ctx.fill();
    
    // 얼굴
    ctx.fillStyle = '#f59e0b';
    ctx.beginPath();
    ctx.arc(x, y - r*0.6, r*0.6, 0, Math.PI*2);
    ctx.fill();
    
    // 귀 (좌우)
    ctx.fillStyle = '#d97706';
    ctx.beginPath();
    ctx.arc(x - r*0.5, y - r*1.1, r*0.25, 0, Math.PI*2);
    ctx.fill();
    ctx.beginPath();
    ctx.arc(x + r*0.5, y - r*1.1, r*0.25, 0, Math.PI*2);
    ctx.fill();
    
    // 눈 (야성적인 붉은 눈)
    ctx.fillStyle = '#dc2626';
    ctx.fillRect(x - r*0.3, y - r*0.7, r*0.15, r*0.1);
    ctx.fillRect(x + r*0.15, y - r*0.7, r*0.15, r*0.1);
    
    // 앞발 공격 애니메이션
    if(isAttacking) {
        ctx.save();
        ctx.translate(x + r*1.2*rotDir, y);
        ctx.fillStyle = '#f59e0b';
        ctx.beginPath();
        ctx.arc(0, 0, r*0.5, 0, Math.PI*2);
        ctx.fill();
        // 발톱
        ctx.fillStyle = '#fef3c7';
        for(let ci = -1; ci <= 1; ci++) {
            ctx.beginPath();
            ctx.arc(ci*r*0.2, r*0.35, r*0.1, 0, Math.PI*2);
            ctx.fill();
        }
        ctx.restore();
    }
    
    // 거대화 오라 (isGiant 상태일 때)
    if(this && this.isGiant) {
        let pulse = 0.6 + Math.sin(performance.now()/200) * 0.3;
        ctx.globalAlpha = pulse;
        ctx.strokeStyle = '#fcd34d';
        ctx.lineWidth = 4;
        // r을 파라미터로 받으므로 이미 거대화된 r값이 들어옴
        ctx.beginPath();
        ctx.arc(x, y, r + 6, 0, Math.PI*2);
        ctx.stroke();
        ctx.globalAlpha = 1;
    }
}
```

### 7-1. 엔티티 반복 최소화

현재 `Hero.update()` 한 호출 당 entities 배열을 평균 5~8회 순회. 10명의 영웅 × 8회 = 프레임당 80회 배열 순회.

**개선**: 자주 사용하는 `nearEnemies` 결과를 캐싱:

```javascript
// Hero.update(dt) 내 상단에 캐시 계산 (한 번만)
if(!this.isDead) {
    // ★ 이 캐시를 한 프레임 내 여러 곳에서 재사용
    this._cachedNearEnemies = entities.filter(e => 
        e.faction !== this.faction && !e.isDead && dist(this,e) < 600
    );
    this._cachedNearAllies = entities.filter(e => 
        e.faction === this.faction && !e.isDead && e !== this && dist(this,e) < 600
    );
}
```

그리고 `handleAI()`, `autoUseHeroSkills()`, `autoAttack()`, `updatePassives()` 내의 `entities.filter(...)` 호출들을 `this._cachedNearEnemies` 로 교체.

### 7-2. Particle 시스템 상한

과도한 파티클 생성으로 후반 FPS 하락. 상한 추가:

```javascript
// spawnParticles 함수 수정
function spawnParticles(x,y,color,n=8,spd=120,life=0.4, shape='circle'){
    if(particles.length > 800) return;  // ★ 상한: 800개 초과 시 생성 안 함
    let spawn = Math.min(n, 800 - particles.length);
    for(let i=0;i<spawn;i++){
        let a=rand(0,Math.PI*2);
        particles.push({x,y,vx:Math.cos(a)*rand(spd*0.3,spd),vy:Math.sin(a)*rand(spd*0.3,spd),life,maxLife:life,color,size:rand(2,5),shape});
    }
}
```

## 8. 신규 제안 사항

### 8-1. 서든데스 크리쳐 미구현 상태

`SUDDEN_DEATH_TIME = 18 * 60`이 설정되어 있고 HTML에 "서든데스" UI가 있지만 실제로 18분이 되어도 게임이 계속 진행됨. 서든데스 구현 필요.

`gameLoop` 내 GS.time 업데이트 이후 추가:

```javascript
// ★ 서든데스 트리거
if(!suddenDeathTriggered && GS.time >= SUDDEN_DEATH_TIME) {
    suddenDeathTriggered = true;
    document.getElementById('suddenDeathWarn').classList.remove('hidden');
    showBanner('⚠️ 서든데스 발동! 넥서스 HP가 빠르게 감소합니다!', '💀', true);
    
    // 서든데스 크리쳐: 양측 본진에서 동시 스폰
    let sdBlue = new Monster(300, 2700, 'boss_dragon');
    sdBlue.faction = 'RED';  // 블루 본진으로 어그로
    sdBlue.maxHp = 15000; sdBlue.hp = sdBlue.maxHp;
    sdBlue.atk = 300; sdBlue.radius = 50;
    entities.push(sdBlue);
    
    let sdRed = new Monster(2700, 300, 'boss_dragon');
    sdRed.faction = 'BLUE';  // 레드 본진으로 어그로
    sdRed.maxHp = 15000; sdRed.hp = sdRed.maxHp;
    sdRed.atk = 300; sdRed.radius = 50;
    entities.push(sdRed);
}
// 서든데스 중: 매 초마다 넥서스 HP 감소
if(suddenDeathTriggered) {
    entities.forEach(e => {
        if(e.type === 'nexus' && !e.isDead) {
            e.hp = Math.max(1, e.hp - e.maxHp * 0.005 * dt);  // 초당 0.5% 감소
        }
    });
}
```

### 8-2. 골드 고블린 미구현

`GOLD_GOBLIN_TIME = 8 * 60`으로 설정되어 있지만 실제로 스폰되지 않음. `gameLoop` 내 dragon spawn 로직 이후에 추가:

```javascript
// 골드 고블린 스폰 (8분 1회)
if(!goblinSpawned && GS.time >= GOLD_GOBLIN_TIME) {
    goblinSpawned = true;
    let gx = 1500 + rand(-300, 300);
    let gy = 1500 + rand(-300, 300);
    let goblin = new Monster(gx, gy, 'goblin');
    goblin.maxHp = 1200; goblin.hp = goblin.maxHp;
    goblin.atk = 10; goblin.moveSpd = 220;  // 매우 빠름
    goblin.radius = 14;
    // 처치 시 거대 골드 보상
    goblin.onDeath = function(attacker) {
        if(attacker && attacker.type === 'hero') {
            attacker.gold += 1500;
            addText(attacker.x, attacker.y-60, '💰 황금 고블린 +1500G!', '#fbbf24', 22);
            spawnSpecial(this.x, this.y, '#fbbf24', 'star', 20, 250, 1.0);
        }
        showBanner('황금 고블린 처치!', '💰', attacker?.faction==='BLUE');
    };
    entities.push(goblin);
    showBanner('황금 고블린 출현! 잡아라!', '💰', true);
}
```

### 8-3. 영웅 선택창 히스토리 기반 개인 특성 표시

// selectHero() 상단에 추가
window.trackHeroSelect = function(heroKey) {

    stats[heroKey] = (stats[heroKey] || 0) + 1;
    localStorage.setItem('avalon_hero_stats', JSON.stringify(stats));
};
