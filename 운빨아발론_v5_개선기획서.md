## 1. 모바일 화면 자동 최적화

### 1-1. 현재 문제점 진단
- `canvas.width = window.innerWidth` 를 **매 프레임마다** 호출 → devicePixelRatio 미적용, 고해상도 기기에서 흐림
- 카카오톡 인앱 브라우저는 `100vh`가 URL바 포함 높이를 반환 → UI가 잘림
- 타이틀 화면이 `overflow-y-auto`인데 `body`에 `overflow:hidden` → 스크롤 불가, 선택 UI 짤림
- 모바일 자동 감지 없음 (수동으로 📱 버튼 눌러야 함)
- 조이스틱 반응 영역이 화면 절반(clientX < innerWidth/2)으로 고정 → 세로 화면에서 너무 좁음

### 1-2. index.html 수정 사항

#### viewport 메타 교체
```html
<!-- 기존 -->
<meta name="viewport" content="width=device-width, initial-scale=1.0, maximum-scale=1.0, user-scalable=no, viewport-fit=cover">

<!-- 교체 -->
<meta name="viewport" content="width=device-width, initial-scale=1.0, maximum-scale=1.0, minimum-scale=1.0, user-scalable=no, viewport-fit=cover">
```

#### body 스타일에 CSS 변수 추가
```css
body, html {
    margin:0; padding:0; width:100%; height:100%;
    overflow:hidden; background:#05070c;
    user-select:none; -webkit-user-select:none; touch-action:none;
    font-family:'Noto Sans KR',system-ui,sans-serif; -webkit-touch-callout:none;
    /* 인앱 브라우저 대응 */
    height: 100dvh; /* dynamic viewport height */
}
```

#### 타이틀 화면 - 세로 모바일 레이아웃 수정
타이틀 화면 div에서 `overflow-y-auto` → `overflow-y-scroll; -webkit-overflow-scrolling: touch` 로 변경하고
내부 컨테이너에 `pb-safe` (iOS safe area) 추가:
```html
<div id="titleScreen" ... style="...padding-bottom: env(safe-area-inset-bottom, 20px);">
```

#### B. Canvas DPI 대응 시스템 (draw 함수 완전 교체)

기존 draw() 함수에서 `canvas.width = window.innerWidth` 하는 부분을 **삭제**하고,
대신 **별도 resizeCanvas 함수**를 만들어 리사이즈 시에만 호출:

```javascript
// 전역에 추가
let canvasDPR = 1;

function resizeCanvas() {
    canvasDPR = Math.min(window.devicePixelRatio || 1, 2); // 최대 2x (성능)
    const w = window.innerWidth;
    const h = window.innerHeight;
    canvas.width  = Math.floor(w * canvasDPR);
    canvas.height = Math.floor(h * canvasDPR);
    canvas.style.width  = w + 'px';
    canvas.style.height = h + 'px';
}

window.addEventListener('resize', resizeCanvas);
window.addEventListener('orientationchange', () => {
    setTimeout(resizeCanvas, 300); // 회전 애니메이션 후 재측정
});

// startGame() 내 requestAnimationFrame 직전에 resizeCanvas() 호출
```

draw() 함수에서 `ctx.save(); ctx.translate(...)` **직전**에 아래 추가:
```javascript
function draw() {
    // DPR 스케일 적용 (기존 canvas.width= 라인 삭제 후 대체)
    ctx.setTransform(1,0,0,1,0,0); // 매트릭스 리셋
    ctx.fillStyle='#0f172a';
    ctx.fillRect(0, 0, canvas.width, canvas.height);

    // DPR 보정 후 게임 좌표계 진입
    ctx.save();
    ctx.scale(canvasDPR, canvasDPR); // ← 이 라인 추가
    ctx.translate(window.innerWidth/2, window.innerHeight/2);
    ctx.scale(camera.zoom, camera.zoom);
    ctx.translate(-camera.x, -camera.y);
    // ... 이하 기존 코드 동일
```

## 2. 사운드 시스템 전면 개편

### 2-1. 현재 문제점
- `triangle/sine` 오실레이터 단음 → 전자음/진저음 느낌
- 타격음이 너무 조용 (gain 0.08)
- hit/shoot/heal/tower 4종류만 존재
- 모바일에서 AudioContext autoplay 정책으로 무음 발생

### 2-2. 개선 방향
**Web Audio API로 실제 게임 타격음 합성** (외부 파일 없이):
- **노이즈 버퍼** + **엔벨로프** = 실제 타격음
- 여러 오실레이터 레이어링 = 묵직한 사운드
- SFX 종류 확대: hit_melee / hit_ranged / hit_critical / skill_burst / skill_magic / level_up / tower_shot / gold_gain / death / ui_click

### 2-3. 전체 교체 코드

기존 `playSFX(type)` 함수를 **완전 삭제**하고 아래로 교체:

```javascript
// ============ 개선된 사운드 시스템 ============
function createNoiseBuffer(ctx, duration) {
    const sampleRate = ctx.sampleRate;
    const buf = ctx.createBuffer(1, sampleRate * duration, sampleRate);
    const data = buf.getChannelData(0);
    for (let i = 0; i < data.length; i++) data[i] = Math.random() * 2 - 1;
    return buf;
}

function playSFX(type) {
    if (!audioCtx) return;
    if (audioCtx.state === 'suspended') { audioCtx.resume(); return; }

    const now = audioCtx.currentTime;
    const master = audioCtx.createGain();
    master.connect(audioCtx.destination);

    const play = (type, freq, endFreq, oType, gainPeak, attack, decay, detune=0) => {
        const osc = audioCtx.createOscillator();
        const g = audioCtx.createGain();
        osc.connect(g); g.connect(master);
        osc.type = oType; osc.frequency.setValueAtTime(freq, now);
        if (detune) osc.detune.setValueAtTime(detune, now);
        if (endFreq) osc.frequency.exponentialRampToValueAtTime(endFreq, now + decay);
        g.gain.setValueAtTime(0, now);
        g.gain.linearRampToValueAtTime(gainPeak, now + attack);
        g.gain.exponentialRampToValueAtTime(0.001, now + attack + decay);
        osc.start(now); osc.stop(now + attack + decay + 0.05);
    };

    const playNoise = (gainPeak, filterFreq, attack, decay, filterType='bandpass') => {
        const buf = createNoiseBuffer(audioCtx, attack + decay + 0.1);
        const src = audioCtx.createBufferSource();
        src.buffer = buf;
        const filter = audioCtx.createBiquadFilter();
        filter.type = filterType; filter.frequency.setValueAtTime(filterFreq, now);
        const g = audioCtx.createGain();
        src.connect(filter); filter.connect(g); g.connect(master);
        g.gain.setValueAtTime(0, now);
        g.gain.linearRampToValueAtTime(gainPeak, now + attack);
        g.gain.exponentialRampToValueAtTime(0.001, now + attack + decay);
        src.start(now); src.stop(now + attack + decay + 0.1);
    };

    if (type === 'hit') {
        // 묵직한 근접 타격음: 저음 펀치 + 노이즈 버스트
        master.gain.setValueAtTime(0.5, now);
        play('osc', 120, 40,  'sine',    0.8, 0.005, 0.08);
        play('osc', 200, 60,  'square',  0.3, 0.003, 0.06);
        playNoise(0.6, 800,   0.003, 0.07, 'bandpass');
    } else if (type === 'hit_critical') {
        // 크리티컬: 더 강렬하고 금속성 느낌
        master.gain.setValueAtTime(0.7, now);
        play('osc', 300, 80,  'sawtooth', 0.6, 0.002, 0.12);
        play('osc', 150, 50,  'sine',     0.9, 0.004, 0.15);
        playNoise(0.8, 2000,  0.002, 0.10, 'highpass');
        playNoise(0.5, 400,   0.004, 0.12, 'bandpass');
    } else if (type === 'shoot') {
        // 화살 발사: 짧고 날카로운 파열음
        master.gain.setValueAtTime(0.35, now);
        play('osc', 600, 200, 'sawtooth', 0.5, 0.002, 0.08);
        playNoise(0.4, 3000,  0.001, 0.05, 'highpass');
    } else if (type === 'skill_burst') {
        // 스킬 폭발: 웅장한 충격파
        master.gain.setValueAtTime(0.6, now);
        play('osc', 80,  20,  'sine',    0.9, 0.01,  0.3);
        play('osc', 200, 50,  'square',  0.4, 0.005, 0.25);
        play('osc', 400, 100, 'sawtooth',0.3, 0.003, 0.2);
        playNoise(0.7, 500,   0.008, 0.25, 'bandpass');
    } else if (type === 'skill_magic') {
        // 마법 스킬: 반짝이는 마법음
        master.gain.setValueAtTime(0.4, now);
        play('osc', 880, 1200,'sine',    0.4, 0.01, 0.2);
        play('osc', 660, 900, 'sine',    0.3, 0.02, 0.25, 700);
        play('osc', 440, 600, 'triangle',0.2, 0.03, 0.3);
    } else if (type === 'heal') {
        // 힐: 상승하는 맑은 음
        master.gain.setValueAtTime(0.25, now);
        play('osc', 440, 660, 'sine',    0.4, 0.02, 0.3);
        play('osc', 550, 770, 'sine',    0.2, 0.04, 0.35, 1200);
    } else if (type === 'tower') {
        // 타워 공격: 묵직한 대포음
        master.gain.setValueAtTime(0.55, now);
        play('osc', 100, 30,  'sine',    0.8, 0.008, 0.2);
        play('osc', 60,  20,  'sine',    0.6, 0.005, 0.25);
        playNoise(0.9, 300,   0.005, 0.18, 'lowpass');
    } else if (type === 'level_up') {
        // 레벨업: 상승하는 팡파레
        master.gain.setValueAtTime(0.4, now);
        [440, 554, 659, 880].forEach((f, i) => {
            const o = audioCtx.createOscillator();
            const g = audioCtx.createGain();
            o.connect(g); g.connect(master);
            o.type = 'sine'; o.frequency.setValueAtTime(f, now + i * 0.08);
            g.gain.setValueAtTime(0, now + i * 0.08);
            g.gain.linearRampToValueAtTime(0.5, now + i * 0.08 + 0.02);
            g.gain.exponentialRampToValueAtTime(0.001, now + i * 0.08 + 0.25);
            o.start(now + i * 0.08); o.stop(now + i * 0.08 + 0.3);
        });
    } else if (type === 'gold') {
        // 골드 획득: 경쾌한 코인음
        master.gain.setValueAtTime(0.25, now);
        play('osc', 1200, 1600, 'sine', 0.4, 0.005, 0.12);
        play('osc', 1600, 2000, 'sine', 0.3, 0.01,  0.1, 1200);
    } else if (type === 'death') {
        // 사망: 하강하는 둔탁한 음
        master.gain.setValueAtTime(0.5, now);
        play('osc', 300, 50,  'sine',    0.7, 0.01, 0.5);
        play('osc', 150, 30,  'sine',    0.5, 0.02, 0.6);
        playNoise(0.4, 200,   0.01, 0.4, 'lowpass');
    } else if (type === 'ui') {
        // UI 클릭
        master.gain.setValueAtTime(0.2, now);
        play('osc', 800, 1200, 'sine', 0.3, 0.005, 0.08);
    }
}
```

## 3. 스킬 이펙트 대폭 강화

### 3-1. 현재 문제점
- 모든 스킬이 `spawnAOE(원형)` + `spawnParticles(동그라미)` 로 동일
- 각 영웅의 스킬이 시각적으로 구분 불가
- 슬래시 이펙트가 너무 단조로움
- 화면 흔들림(Screen Shake) 없음

### 3-2. 새로운 이펙트 시스템 추가

**전역 변수에 추가:**

**유틸 함수 추가 (기존 spawnSlash/spawnAOE 뒤에 삽입):**
```javascript
// 화면 흔들림
function shakeScreen(intensity, duration) {
    screenShake.intensity = intensity;
    screenShake.duration = duration;
}

// 충격파 링 (팽창하다 사라지는 원형)
function spawnRing(x, y, color, maxR=200, life=0.5) {
    ringEffects.push({ x, y, color, r:10, maxR, life, maxLife:life });
}

// 번개 빔 (두 점 사이)
function spawnBeam(x1, y1, x2, y2, color, life=0.2) {
    beamEffects.push({ x1, y1, x2, y2, color, life, maxLife:life,
        segments: Array.from({length:8}, (_,i) => ({ // 지그재그
            ox: rand(-20,20), oy: rand(-20,20)
        }))
    });
}

// 특수 파티클: 별/하트/X 형태
function spawnSpecial(x, y, color, shape='star', n=12, spd=150, life=0.6) {
    for(let i=0;i<n;i++) {
        let a = (Math.PI*2/n)*i;
        particles.push({x,y, vx:Math.cos(a)*spd, vy:Math.sin(a)*spd,
            life, maxLife:life, color, size:rand(3,8), shape});
    }
}
```

**draw() 함수 내 렌더 추가 (slashEffects 렌더 다음):**

```javascript
// 충격파 링 렌더
ringEffects.forEach(re => {
    let alpha = (re.life / re.maxLife) * 0.8;
    ctx.globalAlpha = alpha;
    ctx.strokeStyle = re.color;
    ctx.lineWidth = 4 * (re.life / re.maxLife);
    ctx.beginPath(); ctx.arc(re.x, re.y, re.r, 0, Math.PI*2); ctx.stroke();
    ctx.globalAlpha = 1;
});

// 번개 빔 렌더
beamEffects.forEach(be => {
    let alpha = (be.life / be.maxLife);
    ctx.globalAlpha = alpha;
    ctx.strokeStyle = be.color;
    ctx.lineWidth = 3 * alpha + 1;
    ctx.shadowColor = be.color; ctx.shadowBlur = 10;
    ctx.beginPath(); ctx.moveTo(be.x1, be.y1);
    let dx = be.x2-be.x1, dy = be.y2-be.y1;
    be.segments.forEach((seg,i) => {
        let t=(i+1)/(be.segments.length+1);
        ctx.lineTo(be.x1+dx*t+seg.ox, be.y1+dy*t+seg.oy);
    });
    ctx.lineTo(be.x2, be.y2); ctx.stroke();
    ctx.shadowBlur=0; ctx.globalAlpha=1;
});
```

## 4. AI 고도화

### 4-1. 현재 문제점
- 라인 이동이 단순 좌표 이동 (꺾임 없음)
- 한타 시 팀끼리 뭉치는 로직 없음
- 후퇴 후 회복하면 다시 무조건 돌진
- 타워 포킹 없이 바로 교전
- 정글 AI가 캠프를 못 찾으면 멈춤
- 스킬 사용 안 함 (autoUseHeroSkills는 player에만 동작)

### 4-2. AI State Machine 도입

기존 `handleAI(dt)` 함수를 **완전 교체**:

```javascript
handleAI(dt) {
    // 0. 기본 상태 변수 초기화
    if (!this.aiState) this.aiState = 'LANE';
    if (!this.aiPokingTimer) this.aiPokingTimer = 0;
    if (!this.aiLaneChangeTimer) this.aiLaneChangeTimer = rand(15, 30);

    this.vx = 0; this.vy = 0;

    // 1. 본진 귀환 시 쇼핑
    let myBase = this.faction === 'BLUE' ? {x:300, y:2700} : {x:2700, y:300};
    let distToBase = dist(this, myBase);
    if (distToBase < 400) {
        this.aiShopTimer -= dt;
        if (this.aiShopTimer <= 0) { this.aiShopAI(); this.aiShopTimer = 5; }
    }

    // 2. 스킬 자동 사용 (AI도 사용)
    if (this.heroSkill1Timer <= 0) {
        let near = entities.filter(e=>e.faction!==this.faction&&!e.isDead&&dist(this,e)<=500);
        if (near.length > 0) this.useSkill(1);
    }
    if (this.heroSkill2Timer <= 0) {
        let near = entities.filter(e=>e.faction!==this.faction&&!e.isDead&&dist(this,e)<=500);
        if (near.length > 0) this.useSkill(2);
    }

    // 3. HP 기반 상태 전환
    let hpRatio = this.hp / this.maxHp;
    if (hpRatio <= 0.25 && this.aiState !== 'RETREAT') {
        this.aiState = 'RETREAT';
        this.isRetreating = true;
    }
    if (hpRatio >= 0.85 && this.aiState === 'RETREAT') {
        this.aiState = 'LANE';
        this.isRetreating = false;
    }

    // 4. 상태별 행동
    switch(this.aiState) {

        case 'RETREAT': {
            // 후퇴: 본진으로 도망, 도착 시 멈춤
            if (distToBase < 200) {
                this.vx = 0; this.vy = 0; // 회복 대기
            } else {
                let a = Math.atan2(myBase.y - this.y, myBase.x - this.x);
                this.vx = Math.cos(a) * this.moveSpd * 1.2; // 도망은 빠르게
                this.vy = Math.sin(a) * this.moveSpd * 1.2;
                this.facingDir = myBase.x < this.x ? -1 : 1;
            }
            break;
        }

        case 'TEAMFIGHT': {
            // 한타: 가장 약한 적 집중
            let enemies = entities.filter(e=>e.faction!==this.faction&&e.type==='hero'&&!e.isDead);
            let target = enemies.sort((a,b) => (a.hp/a.maxHp)-(b.hp/b.maxHp))[0]; // 체력 낮은 적 우선
            if (!target || enemies.length === 0) { this.aiState = 'LANE'; break; }
            // 포지션: 근접은 전방, 원거리는 후방 유지
            let isRanged = HERO_TMPL[this.heroKey].type === 'ranged';
            let safeRange = isRanged ? this.range * 0.8 : this.range * 0.7;
            let d = dist(this, target);
            if (d > safeRange) {
                let a = Math.atan2(target.y-this.y, target.x-this.x);
                this.vx = Math.cos(a)*this.moveSpd; this.vy = Math.sin(a)*this.moveSpd;
            } else if (isRanged && d < this.range * 0.5) {
                // 원거리는 너무 가까우면 후퇴
                let a = Math.atan2(this.y-target.y, this.x-target.x);
                this.vx = Math.cos(a)*this.moveSpd*0.7; this.vy = Math.sin(a)*this.moveSpd*0.7;
            } else {
                this.vx = 0; this.vy = 0;
            }
            this.facingDir = target.x < this.x ? -1 : 1;
            // 한타 종료 조건
            if(enemies.length === 0) this.aiState = 'LANE';
            break;
        }

        case 'POKE': {
            // 포킹: 타워 밖에서 사정거리 끝에서 공격
            this.aiPokingTimer -= dt;
            let pokingTarget = entities.find(e=>e.faction!==this.faction&&!e.isDead&&(e.type==='tower'||e.type==='minion')&&dist(this,e)<this.range*1.2);
            if (!pokingTarget || this.aiPokingTimer <= 0) { this.aiState = 'LANE'; break; }
            let d = dist(this, pokingTarget);
            if(d > this.range) {
                let a = Math.atan2(pokingTarget.y-this.y, pokingTarget.x-this.x);
                this.vx = Math.cos(a)*this.moveSpd*0.7; this.vy = Math.sin(a)*this.moveSpd*0.7;
            }
            break;
        }

        case 'JUNGLE': {
            // 정글 사냥
            if (!this.jungleTarget || this.jungleTarget.isDead) {
                let camps = entities.filter(e=>e.type==='jungle'&&!e.isDead);
                this.jungleTarget = camps.sort((a,b)=>dist(this,a)-dist(this,b))[0]; // 가장 가까운 캠프
                if (!this.jungleTarget) { this.aiState = 'LANE'; break; }
            }
            let d = dist(this, this.jungleTarget);
            if (d > 60) {
                let a = Math.atan2(this.jungleTarget.y-this.y, this.jungleTarget.x-this.x);
                this.vx = Math.cos(a)*this.moveSpd; this.vy = Math.sin(a)*this.moveSpd;
            }
            // 라인으로 복귀 조건: 체력 60% 이하이거나 정글 다 클리어
            let remainCamps = entities.filter(e=>e.type==='jungle'&&!e.isDead&&dist(this,e)<600);
            if (hpRatio < 0.6 || remainCamps.length === 0) this.aiState = 'LANE';
            break;
        }

        case 'LANE':
        default: {
            // 상태 전환 체크
            let nearbyEnemyHeroes = entities.filter(e=>e.faction!==this.faction&&e.type==='hero'&&!e.isDead&&dist(this,e)<500);
            let nearbyAlliedHeroes = entities.filter(e=>e.faction===this.faction&&e.type==='hero'&&!e.isDead&&e!==this&&dist(this,e)<400);

            // 한타 조건: 적 영웅 발견 + 아군도 근처에 있음
            if (nearbyEnemyHeroes.length > 0 && nearbyAlliedHeroes.length >= 1) {
                this.aiState = 'TEAMFIGHT'; break;
            }

            // 포킹 조건: 원거리 영웅이 적 타워 근처에서
            if (HERO_TMPL[this.heroKey].type === 'ranged' && nearbyEnemyHeroes.length === 0) {
                let nearTower = entities.find(e=>e.faction!==this.faction&&(e.type==='tower'||e.type==='nexus_turret')&&!e.isDead&&dist(this,e)<this.range*1.5);
                if (nearTower && Math.random() < 0.3) {
                    this.aiState = 'POKE'; this.aiPokingTimer = rand(3,6); break;
                }
            }

            // 정글 조건: 30% 확률로 정글 전환 (역할이 jungle인 경우)
            if (this.laneRole === 'jungle') { this.aiState = 'JUNGLE'; break; }

            // 라인 변경 타이머
            this.aiLaneChangeTimer -= dt;
            if (this.aiLaneChangeTimer <= 0) {
                let roles = ['top', 'mid', 'bot', 'jungle'];
                // 팀 구성 고려: 같은 라인에 너무 몰리지 않게
                let roleCount = {};
                entities.filter(e=>e.faction===this.faction&&e.type==='hero'&&!e.isDead).forEach(e=>{ roleCount[e.laneRole] = (roleCount[e.laneRole]||0)+1; });
                let bestRole = roles.sort((a,b)=>(roleCount[a]||0)-(roleCount[b]||0))[0];
                this.laneRole = bestRole;
                this.aiLaneChangeTimer = rand(20, 40);
            }

            // 라인 이동 목표 계산
            let enemyBase = this.faction === 'BLUE' ? {x:2700, y:300} : {x:300, y:2700};
            let tx, ty;

            // 라인별 웨이포인트 (3라인 구조 반영)
            if (this.laneRole === 'top') {
                // 탑 라인: 왼쪽 위 경로
                if (this.faction === 'BLUE') {
                    tx = this.x > 400 ? 300 : 300;
                    ty = this.y < 800 ? 300 : Math.max(300, this.y - 100);
                } else {
                    tx = this.x < 2600 ? 2700 : 2700;
                    ty = this.y > 2200 ? 2700 : Math.min(2700, this.y + 100);
                }
            } else if (this.laneRole === 'bot') {
                // 바텀 라인: 오른쪽 아래 경로
                if (this.faction === 'BLUE') {
                    tx = this.y > 2600 ? enemyBase.x : Math.min(2700, this.x + 100);
                    ty = 2700;
                } else {
                    tx = this.y < 400 ? enemyBase.x : Math.max(300, this.x - 100);
                    ty = 300;
                }
            } else {
                // 미드 라인: 대각선
                let progress = this.faction === 'BLUE' ?
                    (this.x + (MAP_SIZE - this.y)) / (MAP_SIZE * 2) :
                    ((MAP_SIZE - this.x) + this.y) / (MAP_SIZE * 2);
                let midProgress = clamp(progress + 0.05, 0, 1);
                tx = this.faction === 'BLUE' ? midProgress * MAP_SIZE : (1-midProgress) * MAP_SIZE;
                ty = this.faction === 'BLUE' ? (1-midProgress) * MAP_SIZE : midProgress * MAP_SIZE;
            }

            // 타워 다이브 방지
            let underEnemyTower = entities.some(t =>
                (t.type==='tower'||t.type==='nexus_turret') &&
                t.faction!==this.faction && !t.isDead && dist(t,this) < t.range+50
            );
            if (underEnemyTower && hpRatio < 0.7) {
                // 타워 아래에서 도망
                let a = Math.atan2(myBase.y-this.y, myBase.x-this.x);
                this.vx = Math.cos(a)*this.moveSpd; this.vy = Math.sin(a)*this.moveSpd;
                break;
            }

            // 목표 방향 이동
            let dToTarget = dist(this, {x:tx, y:ty});
            if (dToTarget > 30) {
                let a = Math.atan2(ty-this.y, tx-this.x);
                this.vx = Math.cos(a)*this.moveSpd; this.vy = Math.sin(a)*this.moveSpd;
                this.facingDir = tx < this.x ? -1 : 1;
            }
            break;
        }
    }
}
```
